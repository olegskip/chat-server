CREATE TABLE "users" (
	"id"	INTEGER PRIMARY KEY AUTOINCREMENT UNIQUE,
	"email"	INTEGER,
	"login"	TEXT NOT NULL UNIQUE,
	"password"	TEXT NOT NULL UNIQUE,
	"isOnline"	INTEGER DEFAULT 0,
	"isLogIn"	INTEGER DEFAULT 0,
	"color"	TEXT DEFAULT '#ffffff',
	"ip"	TEXT DEFAULT 0,
	"isAdmin"	INTEGER DEFAULT 0
)
